Using

This component is available via maven. Include in your pom.xml file the following :-

    <project>
        ...
        <dependencies>
            <dependency>
                <groupId>com.tibco.ep.samples.docker</groupId>
                <artifactId>lv-1node-lv</artifactId>
                <type>ep-liveview-fragment</type>
            </dependency>
            ...
        </dependencies>
        ...
        <dependencyManagement>
            <dependencies>
                <dependency>
                     <groupId>com.tibco.ep.samples.docker</groupId>
                     <artifactId>lv-1node-lv</artifactId>
                     <version>1.0.0</version>
                 </dependency>
                 ...
             </dependencies>
        </dependencyManagement>
        ...
    </project>
